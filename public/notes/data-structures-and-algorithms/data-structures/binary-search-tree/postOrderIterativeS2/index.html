postOrderIterativeS2()

Binary Search Tree Assignment

Write an iterative C function postOrderIterativeS2() that prints the post-order traversal of a binary search tree using two stacks. Note that you should only use push() or pop() operations when you add or remove integers from the stacks. Remember to empty the stacks at the beginning, if the stacks are not empty.

The function prototype is given as follows:

void postOrderIterativeS2(BSTNode *root);

Following is the detailed algorithm:

1) Push root to first stack.
2) Loop while first stack is not empty
    2.1) Pop a node from first stack and push it to second stack
    2.2) Push left and right children of the popped node to first stack
3) Print contents of second stack

Let us consider the below tree for example.


Iterative Postorder Traversal: 10 18 15 25 80 50 20

void postOrderIterativeS2(BSTNode *root)
{
	 /* add your code here */
	 Stack s1;
	 Stack s2;
	 s1.top = NULL;
	 s2.top = NULL;
	 BSTNode *temp = root;

	 if (temp != NULL)
     {
         push(&s1,temp);
         while(!isEmpty(&s1)){
            temp = pop(&s1);
            push(&s2,temp);
            if (temp->right!=NULL)
                push(&s1,temp->right);
            if (temp->left!=NULL)
                push(&s1,temp->left);
         }
         while(!isEmpty(&s2)){
            temp = pop(&s2);
            printf("%d ",temp->item);
         }
     }
}

void insertBSTNode(BSTNode **node, int value){
	if (*node == NULL)
	{
		*node = malloc(sizeof(BSTNode));

		if (*node != NULL) {
			(*node)->item = value;
			(*node)->left = NULL;
			(*node)->right = NULL;
		}
	}
	else
	{
		if (value < (*node)->item)
		{
			insertBSTNode(&((*node)->left), value);
		}
		else if (value >(*node)->item)
		{
			insertBSTNode(&((*node)->right), value);
		}
		else
			return;
	}
}

void push(Stack *stack, BSTNode * node)
{
	StackNode *temp;

	temp = malloc(sizeof(StackNode));

	if (temp == NULL)
		return;
	temp->data = node;

	if (stack->top == NULL)
	{
		stack->top = temp;
		temp->next = NULL;
	}
	else
	{
		temp->next = stack->top;
		stack->top = temp;
	}
}


BSTNode * pop(Stack * s)
{
	StackNode *temp, *t;
	BSTNode * ptr;
	ptr = NULL;

	t = s->top;
	if (t != NULL)
	{
		temp = t->next;
		ptr = t->data;

		s->top = temp;
		free(t);
		t = NULL;
	}

	return ptr;
}

BSTNode * peek(Stack * s)
{
	StackNode *temp;
	temp = s->top;
	if (temp != NULL)
		return temp->data;
	else
		return NULL;
}

int isEmpty(Stack *s)
{
	if (s->top == NULL)
		return 1;
	else
		return 0;
}


void removeAll(BSTNode **node)
{
	if (*node != NULL)
	{
		removeAll(&((*node)->left));
		removeAll(&((*node)->right));
		free(*node);
		*node = NULL;
	}
}
