int smallestValue(BTNode *node)
{
	int l,r;
	if (node==NULL)
		return 100000;

	l=smallestValue(node->left);
	r=smallestValue(node->right);
	if (l<node->item && l<r)
		return l;
	else if (r<node->item && r<l)
		return r;
	else return node->item;
}
