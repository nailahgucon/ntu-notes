Dual Search - Sorted Array

Hashing Lab Dual Search - Unsorted Array

Given a sorted array of n elements. Find two elements in the array such that their sum is equal to K. The two elements can be the same element. Once a pair of elements are found, the program can be terminated. The results may be different from the results of Question 1.

The function prototype is given as follows:

void dualSortedSearch (int[],int, int, int[]); 

#include <stdio.h>
#include <stdlib.h>

void quickSort(int A[], int low, int high);
int partition (int A[], int low, int high);

void dualSortedSearch (int[],int, int, int[]);

int main()
{
    int i,size,K;
    int* A;
    int index[2] = {-1,-1};

    //Search key
    scanf("%d",&K);

    //create an array
    scanf("%d",&size);
    A=(int *)malloc(sizeof(int)*size);
    for(i=0; i<size; i++)
    {
        scanf("%d",&A[i]);
    }

    quickSort(A,0,size-1);
    dualSortedSearch(A,size,K,index);

    if(index[0]!=-1)
        printf("%d %d\n",index[0], index[1]);
    else
        printf("Not found");

    free(A);
    return 0;
}

void dualSortedSearch(int A[], int size, int K, int dualIndex[])
{
    int i,j, temp;

    i=0;
    j=size-1;
    while(i<j){
        temp = A[i]+A[j];

        if(temp == K)
        {
            dualIndex[0] = i;
            dualIndex[1] = j;
            return;
        }
        else if(temp<K)
        {
            i++;
        }
        else
            j--;
    }
}

int partition (int A[], int low, int high)
{
    int pivot = A[high];
    int i = (low - 1);

    int temp;
    for (int j = low; j <= high- 1; j++)
    {
        if (A[j] < pivot)
        {
            i++;

            temp = A[i];
            A[i] = A[j];
            A[j] = temp;
        }
    }

    temp = A[i+1];
    A[i+1]= A[high];
    A[high] = temp;
    return (i + 1);
}

void quickSort(int A[], int low, int high)
{
    if (low < high)
    {
        int pivot = partition(A, low, high);

        quickSort(A, low, pivot - 1);
        quickSort(A, pivot + 1, high);
    }
}
