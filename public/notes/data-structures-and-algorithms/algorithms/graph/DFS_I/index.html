DFS_I()

Graph Lab

Write a function DFS_I() to do a depth search from a input vertex v. The labels of v are from 1 to |V|. The algorithm will visit the neighbor nodes in ascending order.

The function prototype is given as follows:

void DFS_I ( Graph g, int v);

The structure of a graph is given below:

typedef struct _listnode {
    int vertex;
    struct _listnode * next;
} ListNode;
typedef ListNode StackNode;

typedef struct _graph {
    int V;
    int E;
    int ** matrix;
} Graph;

typedef struct _linkedlist {
    int size;
    ListNode * head;
} Stack;

A test sample graph and its expected output are given below: The start vertex for DFS is vertex 11.



The expected output: 5 9 10 6 2 7 12 8 3 4 1 11

void DFS_I (Graph g, int v)
{
    Stack s;
    s.size = 0;
    s.head = NULL;

    int i;
    int w;

    int *visitV;
    visitV = (int *) malloc(sizeof(int)*g.V);
    for(i=0;i<g.V;i++) visitV[i] = 0;

    push(&s,v);
    visitV[v-1] = 1;

    int unvisited; // no unvisited vertices flag
    while(isEmptyStack(s)==0){
        w = peek(s);

        unvisited = 0;
        for(i=0;i<g.V;i++)
        {
            if(g.matrix[w-1][i] == 1 && visitV[i]==0)
            {
                push(&s,i+1);
                visitV[i]=1;
                unvisited = 1;
                break;
            }
        }

        if(unvisited==0){
           printf("%d ",w);
           pop(&s);
        }
    }

    free(visitV);
}

void printGraphMatrix(Graph g)
{
    int i,j;

    for(i=0;i<g.V;i++){
        for(j=0;j<g.V;j++)
            printf("%d\t",g.matrix[i][j]);
        printf("\n");
    }

}

void push(Stack *sPtr, int vertex)
{
	StackNode *newNode;
    newNode= malloc(sizeof(StackNode));
    newNode->vertex = vertex;
    newNode->next = sPtr->head;
    sPtr->head = newNode;
    sPtr->size++;
}

int pop(Stack *sPtr)
{
    if(sPtr==NULL || sPtr->head==NULL){
        return 0;
    }
    else{
       StackNode *temp = sPtr->head;
       sPtr->head = sPtr->head->next;
       free(temp);
       sPtr->size--;
       return 1;
    }
}

int isEmptyStack(Stack s)
{
     if(s.size==0) return 1;
     else return 0;
}

int peek(Stack s){
    return s.head->vertex;
}

void removeAllItemsFromStack(Stack *sPtr)
{
	while(pop(sPtr));
