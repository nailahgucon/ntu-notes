CircleIntersect() 
// Intersect()
// Parameters: struct circle c1, struct circle c2

double a, b;
int result;

a = c1.x - c2.x;
b = c1.y - c2.y;

return (sqrt(a*a + b*b) <= (c1.radius + c2.radius));

// Contain()
// Parameters: struct circle *c1, struct *circle c2

double a, b;

a = c1->x - c2->x;
b = c1->y - c2->y;

return (c1->radius >= (c2->radius + sqrt(a*a + b*b)));

    case '-':

        break;
