int i,j=0;
int len = strlen(str);
for(i=len-1;i>=0;i--){
    newstr[j++] = str[i];

        max=str[i];
        maxindex=i;

LongWordLength()

// Parameters: *s
// include ctype.h, string.h

int longest = 0;
int current  = 0;

while(*s != '\0'){
    if(isalpha(*s)){
        current++;
    }else{
        current = 0;
    }
    if(current > longest){
        longest = current;
    }s++;
    return longest;
}
